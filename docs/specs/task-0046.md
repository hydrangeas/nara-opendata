```tsx
// src/presentation/components/auth/ProtectedRoute.tsx
import React, { useEffect, useState } from "react";
import { Navigate, useLocation } from "react-router-dom";
import { useAuth } from "@/presentation/hooks/useAuth";
import { Loader2 } from "lucide-react";

interface ProtectedRouteProps {
  children: React.ReactNode;
  requiredTier?: string[];
  fallbackPath?: string;
}

export const ProtectedRoute: React.FC<ProtectedRouteProps> = ({
  children,
  requiredTier,
  fallbackPath = "/login",
}) => {
  const { user, session, loading, isAuthenticated } = useAuth();
  const location = useLocation();
  const [isValidating, setIsValidating] = useState(true);

  useEffect(() => {
    validateAccess();
  }, [session, user]);

  const validateAccess = async () => {
    if (!loading) {
      setIsValidating(false);
    }
  };

  // ローディング中
  if (loading || isValidating) {
    return (
      <div className="flex min-h-screen items-center justify-center">
        <div className="text-center">
          <Loader2 className="h-8 w-8 animate-spin text-primary mx-auto" />
          <p className="mt-2 text-sm text-gray-600">認証状態を確認中...</p>
        </div>
      </div>
    );
  }

  // 未認証
  if (!isAuthenticated) {
    // 現在のパスを保存してログイン後にリダイレクト
    const redirectUrl = `${location.pathname}${location.search}`;
    return (
      <Navigate 
        to={fallbackPath} 
        state={{ from: redirectUrl }} 
        replace 
      />
    );
  }

  // ティアチェック
  if (requiredTier && requiredTier.length > 0) {
    const userTier = user?.app_metadata?.tier || "tier1";
    
    if (!requiredTier.includes(userTier)) {
      return (
        <div className="flex min-h-screen items-center justify-center">
          <div className="text-center max-w-md">
            <h2 className="text-2xl font-bold text-gray-900 mb-2">
              アクセス権限がありません
            </h2>
            <p className="text-gray-600 mb-4">
              このページにアクセスするには、より高いティアのプランが必要です。
            </p>
            <a
              href="/settings/billing"
              className="inline-flex items-center px-4 py-2 border border-transparent text-sm font-medium rounded-md shadow-sm text-white bg-indigo-600 hover:bg-indigo-700"
            >
              プランをアップグレード
            </a>
          </div>
        </div>
      );
    }
  }

  // 認証済みでアクセス権限あり
  return <>{children}</>;
};
```

### 認証ガードフック
```typescript
// src/presentation/hooks/useAuthGuard.ts
import { useEffect, useCallback } from "react";
import { useNavigate, useLocation } from "react-router-dom";
import { useAuth } from "./useAuth";
import { supabase } from "@/infrastructure/supabase/client";

interface UseAuthGuardOptions {
  redirectTo?: string;
  requiredTier?: string[];
  onUnauthorized?: () => void;
}

export const useAuthGuard = (options: UseAuthGuardOptions = {}) => {
  const { 
    redirectTo = "/login", 
    requiredTier,
    onUnauthorized 
  } = options;
  
  const { isAuthenticated, user, loading } = useAuth();
  const navigate = useNavigate();
  const location = useLocation();

  const checkAuth = useCallback(async () => {
    if (loading) return;

    if (!isAuthenticated) {
      // 未認証の場合
      const returnUrl = `${location.pathname}${location.search}`;
      navigate(redirectTo, { 
        state: { from: returnUrl },
        replace: true 
      });
      return;
    }

    // ティアチェック
    if (requiredTier && requiredTier.length > 0) {
      const userTier = user?.app_metadata?.tier || "tier1";
      
      if (!requiredTier.includes(userTier)) {
        if (onUnauthorized) {
          onUnauthorized();
        } else {
          navigate("/unauthorized", { replace: true });
        }
      }
    }
  }, [isAuthenticated, user, loading, location, navigate, redirectTo, requiredTier, onUnauthorized]);

  useEffect(() => {
    checkAuth();
  }, [checkAuth]);

  // セッション更新の監視
  useEffect(() => {
    const checkSession = async () => {
      const { data: { session }, error } = await supabase.auth.getSession();
      
      if (error || !session) {
        // セッション無効
        const returnUrl = `${location.pathname}${location.search}`;
        navigate(redirectTo, { 
          state: { from: returnUrl },
          replace: true 
        });
      }
    };

    // 5分ごとにセッションをチェック
    const interval = setInterval(checkSession, 5 * 60 * 1000);

    return () => clearInterval(interval);
  }, [location, navigate, redirectTo]);

  return {
    isAuthenticated,
    isLoading: loading,
    user,
  };
};
```

```tsx
          path="/dashboard"
          element={

              <DashboardPage />
            </ProtectedRoute>
          }
        />
        
        {/* 管理者専用ルート */}
        <Route
          path="/admin/*"
          element={
            <ProtectedRoute requiredTier={["tier3"]}>

          }
        />
        
        {/* エラーページ */}
        <Route path="/unauthorized" element={<UnauthorizedPage />} />
        <Route path="/404" element={<NotFoundPage />} />
        <Route path="*" element={<Navigate to="/404" replace />} />
      </Routes>
    </BrowserRouter>
  );
};
```

### セッション管理サービス
```typescript
// src/infrastructure/auth/session-manager.ts
import { injectable } from "tsyringe";
import { supabase } from "@/infrastructure/supabase/client";
import { Session } from "@supabase/supabase-js";

@injectable()
export class SessionManager {
  private refreshTimer?: NodeJS.Timeout;
  private readonly REFRESH_INTERVAL = 30 * 60 * 1000; // 30分
  private readonly WARNING_THRESHOLD = 5 * 60 * 1000; // 5分

  startSessionRefresh(onExpiringSoon?: () => void): void {
    this.stopSessionRefresh();

    this.refreshTimer = setInterval(async () => {
      const session = await this.getSession();
      
      if (!session) {
        this.stopSessionRefresh();
        return;
      }

      const expiresAt = session.expires_at ? session.expires_at * 1000 : 0;
      const now = Date.now();
      const timeUntilExpiry = expiresAt - now;

      // 有効期限が近い場合は警告
      if (timeUntilExpiry < this.WARNING_THRESHOLD && onExpiringSoon) {
        onExpiringSoon();
      }

      // セッションをリフレッシュ
      if (timeUntilExpiry < this.REFRESH_INTERVAL) {
        await this.refreshSession();
      }
    }, 60 * 1000); // 1分ごとにチェック
  }

  stopSessionRefresh(): void {
    if (this.refreshTimer) {
      clearInterval(this.refreshTimer);
      this.refreshTimer = undefined;
    }
  }

  async getSession(): Promise<Session | null> {
    const { data: { session }, error } = await supabase.auth.getSession();
    
    if (error) {
      console.error("Failed to get session:", error);
      return null;
    }

    return session;
  }

  async refreshSession(): Promise<Session | null> {
    const { data: { session }, error } = await supabase.auth.refreshSession();
    
    if (error) {
      console.error("Failed to refresh session:", error);
      return null;
    }

    return session;
  }

  async validateSession(): Promise<boolean> {
    const session = await this.getSession();
    
    if (!session) {
      return false;
    }

    // トークンの有効期限を確認
    const expiresAt = session.expires_at ? session.expires_at * 1000 : 0;
    const isExpired = Date.now() > expiresAt;

    return !isExpired;
  }
}
```

```typescript
import { render, screen, waitFor } from "@testing-library/react";

        </ProtectedRoute>
      </MemoryRouter>
    );

    await waitFor(() => {
      expect(screen.getByText("Protected Content")).toBeInTheDocument();
```
