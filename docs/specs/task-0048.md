### ログアウトボタンコンポーネント
```tsx
// src/presentation/components/auth/LogoutButton.tsx
import React, { useState } from "react";
import { useNavigate } from "react-router-dom";
import { useAuth } from "@/presentation/hooks/useAuth";
import { Button } from "@/components/ui/button";
import {
  AlertDialog,
  AlertDialogAction,
  AlertDialogCancel,
  AlertDialogContent,
  AlertDialogDescription,
  AlertDialogFooter,
  AlertDialogHeader,
  AlertDialogTitle,
  AlertDialogTrigger,
} from "@/components/ui/alert-dialog";
import { LogOut, Loader2 } from "lucide-react";
import { toast } from "@/components/ui/use-toast";

interface LogoutButtonProps {
  variant?: "default" | "outline" | "ghost";
  size?: "default" | "sm" | "lg" | "icon";
  showIcon?: boolean;
  showText?: boolean;
  onLogoutComplete?: () => void;
}

export const LogoutButton: React.FC<LogoutButtonProps> = ({
  variant = "ghost",
  size = "default",
  showIcon = true,
  showText = true,
  onLogoutComplete,
}) => {
  const navigate = useNavigate();
  const { signOut } = useAuth();
  const [isLoggingOut, setIsLoggingOut] = useState(false);
  const [showDialog, setShowDialog] = useState(false);

  const handleLogout = async () => {
    try {
      setIsLoggingOut(true);

      // ログアウトイベントをサーバーに送信
      try {
        await fetch("/api/auth/logout", {
          method: "POST",
          headers: {
            "Content-Type": "application/json",
            "Authorization": `Bearer ${localStorage.getItem("access_token")}`,
          },
        });
      } catch (error) {
        // ログ記録の失敗は無視（ログアウト自体は続行）
        console.error("Failed to record logout event:", error);
      }

      // Supabaseからログアウト
      await signOut();

      // ローカルストレージをクリア
      clearLocalStorage();

      // セッションストレージもクリア
      sessionStorage.clear();

      // キャッシュをクリア
      if ('caches' in window) {
        const cacheNames = await caches.keys();
        await Promise.all(
          cacheNames.map(cacheName => caches.delete(cacheName))
        );
      }

      // 成功メッセージ
      toast({
        title: "ログアウトしました",
        description: "ご利用ありがとうございました",
      });

      // コールバック実行
      if (onLogoutComplete) {
        onLogoutComplete();
      }

      // ログインページにリダイレクト
      navigate("/login", { replace: true });
    } catch (error) {
      console.error("Logout error:", error);
      
      toast({
        title: "ログアウトエラー",
        description: "ログアウト中にエラーが発生しました",
        variant: "destructive",
      });

      // エラーが発生してもログインページに移動
      navigate("/login", { replace: true });
    } finally {
      setIsLoggingOut(false);
      setShowDialog(false);
    }
  };

  const clearLocalStorage = () => {
    // 保持したいキー（例：ユーザー設定）
    const keysToKeep = ["theme", "language"];
    
    // 保持するデータを一時保存
    const dataToKeep: Record<string, string> = {};
    keysToKeep.forEach(key => {
      const value = localStorage.getItem(key);
      if (value) {
        dataToKeep[key] = value;
      }
    });

    // すべてクリア
    localStorage.clear();

    // 保持するデータを復元
    Object.entries(dataToKeep).forEach(([key, value]) => {
      localStorage.setItem(key, value);
    });
  };

  return (
    <AlertDialog open={showDialog} onOpenChange={setShowDialog}>
      <AlertDialogTrigger asChild>
        <Button
          variant={variant}
          size={size}
          disabled={isLoggingOut}
          className={showText ? "" : "px-2"}
        >
          {isLoggingOut ? (
            <Loader2 className="h-4 w-4 animate-spin" />
          ) : (
            <>
              {showIcon && <LogOut className={showText ? "mr-2 h-4 w-4" : "h-4 w-4"} />}
              {showText && "ログアウト"}
            </>
          )}
        </Button>
      </AlertDialogTrigger>
      <AlertDialogContent>
        <AlertDialogHeader>
          <AlertDialogTitle>ログアウトの確認</AlertDialogTitle>
          <AlertDialogDescription>
            本当にログアウトしますか？再度APIを利用するにはログインが必要です。
          </AlertDialogDescription>
        </AlertDialogHeader>
        <AlertDialogFooter>
          <AlertDialogCancel disabled={isLoggingOut}>
            キャンセル
          </AlertDialogCancel>
          <AlertDialogAction
            onClick={handleLogout}
            disabled={isLoggingOut}
            className="bg-red-600 hover:bg-red-700 focus:ring-red-600"
          >
            {isLoggingOut ? (
              <>
                <Loader2 className="mr-2 h-4 w-4 animate-spin" />
                ログアウト中...
              </>
            ) : (
              "ログアウト"
            )}
          </AlertDialogAction>
        </AlertDialogFooter>
      </AlertDialogContent>
    </AlertDialog>
  );
};
```

### ログアウトAPIエンドポイント
```typescript
// src/presentation/api/auth/logout.route.ts
import { FastifyPluginAsync } from "fastify";
import { container } from "tsyringe";
import { AuthEventHandler } from "@/infrastructure/auth/auth-event-handler";

export const logoutRoute: FastifyPluginAsync = async (fastify) => {
  fastify.post("/api/auth/logout", {
    preHandler: fastify.authenticate,
    handler: async (request, reply) => {
      try {
        const authEventHandler = container.resolve(AuthEventHandler);
        
        // ログアウトイベントを記録
        await authEventHandler.handleAuthEvent(
          "SIGNED_OUT",
          null,
          {
            ipAddress: request.ip,
            userAgent: request.headers["user-agent"] || "Unknown",
            provider: "manual",
            metadata: {
              userId: request.user?.userId.value,
              logoutType: "user_initiated",
            },
          }
        );

        reply.code(200).send({ success: true });
      } catch (error) {
        request.log.error({ error }, "Failed to record logout event");
        // ログ記録の失敗はクライアントに影響させない
        reply.code(200).send({ success: true });
      }
    },
  });
};
```
